Forward proof with tree

let th1 = assume_rule_fd p
list_gen_rule_fd [p;q]
  (deduct_antisym_rule_fd
    (conj_rule_fd th1 (disj1_rule_fd th1 q))
    (conjunct1_rule_fd (assume_rule_fd (parse_term(@"p /\ (p \/ q)")))) )
|> zipper
|> view
