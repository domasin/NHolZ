|> start_proof
(* |- true                        *)
|> eq_mp_rule_bk ("(\(p:bool). p) = \p. p" |> mkGoal [])
    (* |- (\p. p) = (\p. p) <=> true  *)
    |> sym_rule_bk
        (* |- true <=> (\(p:bool). p) = (\p. p) *)
        |> by true_def "true\_def"                          |> lower |> prove |> right
    (* |- (\p. p) = (\p. p)           *)
    |> refl_conv_bk                                         |> lower |> prove |> lower |> prove
|> view
|> loc_thm |> Option.get
